Pattern matching

Destructuring


(def some-numbers [1 2 3 4 5])

(let [[_ _ _ x _] some-numbers] x)
; => 4

Fizz Buzz example


  (use '[clojure.core.match :only (match)])

  (doseq [n (range 1 101)]
    (println
      (match [(mod n 3) (mod n 5)]
        [0 0] "FizzBuzz"
        [0 _] "Fizz"
        [_ 0] "Buzz"
        :else n)))

Macros


(unless (> 1 2)
  "this will be returned"
  "this won't'")

(defmacro unless
  [test-exp then else]
  `(if (not ~test-exp)
     ~then
     ~else))
